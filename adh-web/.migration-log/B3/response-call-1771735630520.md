```typescript
import { apiClient, type ApiResponse } from './apiClient';
import type {
  HistoriqueAppel,
  CoefTelephone,
  ReseauCloture,
  FacturationRequest,
  FacturationResponse,
  GetHistoriqueAppelsResponse,
  GetCoefficientResponse,
  FacturerAppelResponse,
  GetClotureStatusResponse,
  DebloquerClotureResponse,
  AnnulerFacturationResponse,
} from '@/types/facturationAppel';

export const facturationAppelApi = {
  getHistoriqueAppels: (
    societe: string,
    compte: string,
    filiation: string,
    dateDebut?: string,
    dateFin?: string,
    numeroTel?: string,
  ) => {
    const params = new URLSearchParams();
    params.append('societe', societe);
    params.append('compte', compte);
    params.append('filiation', filiation);
    if (dateDebut) params.append('dateDebut', dateDebut);
    if (dateFin) params.append('dateFin', dateFin);
    if (numeroTel) params.append('numeroTel', numeroTel);

    return apiClient.get<GetHistoriqueAppelsResponse>(
      `/telephone/historique?${params.toString()}`,
    );
  },

  getCoefficientTelephone: (qualite: string) =>
    apiClient.get<GetCoefficientResponse>(
      `/telephone/coefficient/${encodeURIComponent(qualite)}`,
    ),

  facturationAppel: (data: FacturationRequest) =>
    apiClient.post<FacturerAppelResponse>(
      '/telephone/facturer',
      data,
    ),

  annulerFacturation: (appelId: number) =>
    apiClient.post<AnnulerFacturationResponse>(
      `/telephone/annuler/${appelId}`,
      {},
    ),

  getReseauStatus: () =>
    apiClient.get<GetClotureStatusResponse>(
      '/reseau/status',
    ),

  debloquerReseau: () =>
    apiClient.post<DebloquerClotureResponse>(
      '/reseau/debloquer',
      {},
    ),
};
```
