```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { usePrintConfigStore } from '@/stores/printConfigurationStore';
import { useDataSourceStore } from '@/stores/dataSourceStore';
import { apiClient } from '@/services/api/apiClient';
import type { ApiResponse } from '@/services/api/apiClient';
import type { PrintConfig } from '@/types/printConfiguration';

vi.mock('@/services/api/apiClient', () => ({
  apiClient: {
    get: vi.fn(),
    post: vi.fn(),
  },
}));

const MOCK_CONFIG: PrintConfig = {
  currentListingNum: 42,
  currentPrinterName: 'HP_LaserJet',
  currentPrinterNum: 1,
  numberCopies: 2,
  specificPrint: 'CUSTOM',
};

const DEFAULT_CONFIG: PrintConfig = {
  currentListingNum: 0,
  currentPrinterName: 'VOID',
  currentPrinterNum: 0,
  numberCopies: 0,
  specificPrint: 'VOID',
};

describe('printConfigurationStore', () => {
  beforeEach(() => {
    usePrintConfigStore.getState().reset();
    useDataSourceStore.setState({ isRealApi: false });
    vi.clearAllMocks();
  });

  describe('setListingNumber', () => {
    it('should set listing number and initialize default print parameters in mock mode', async () => {
      const listingNumber = 123;

      await usePrintConfigStore.getState().setListingNumber(listingNumber);

      const state = usePrintConfigStore.getState();
      expect(state.currentConfig).toEqual({
        currentListingNum: listingNumber,
        currentPrinterName: 'VOID',
        currentPrinterNum: 0,
        numberCopies: 0,
        specificPrint: 'VOID',
      });
      expect(state.isInitializing).toBe(false);
      expect(state.error).toBeNull();
    });

    it('should call API and update config in real API mode', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      const listingNumber = 456;

      vi.mocked(apiClient.post).mockResolvedValueOnce({
        data: { success: true, data: undefined },
      } as ApiResponse<void>);

      vi.mocked(apiClient.get).mockResolvedValueOnce({
        data: { success: true, data: MOCK_CONFIG },
      } as ApiResponse<PrintConfig>);

      await usePrintConfigStore.getState().setListingNumber(listingNumber);

      expect(apiClient.post).toHaveBeenCalledWith('/api/print-config/set-listing', {
        listingNumber,
      });
      expect(apiClient.get).toHaveBeenCalledWith('/api/print-config');
      expect(usePrintConfigStore.getState().currentConfig).toEqual(MOCK_CONFIG);
      expect(usePrintConfigStore.getState().error).toBeNull();
    });

    it('should set error and throw on API failure', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      const errorMessage = 'Failed to set listing number';

      vi.mocked(apiClient.post).mockRejectedValueOnce(new Error(errorMessage));

      await expect(
        usePrintConfigStore.getState().setListingNumber(789)
      ).rejects.toThrow(errorMessage);

      const state = usePrintConfigStore.getState();
      expect(state.error).toBe(errorMessage);
      expect(state.isInitializing).toBe(false);
    });

    it('should set isInitializing during operation', async () => {
      let isInitializing = false;

      vi.mocked(apiClient.post).mockImplementationOnce(async () => {
        isInitializing = usePrintConfigStore.getState().isInitializing;
        return { data: { success: true, data: undefined } } as ApiResponse<void>;
      });

      vi.mocked(apiClient.get).mockResolvedValueOnce({
        data: { success: true, data: MOCK_CONFIG },
      } as ApiResponse<PrintConfig>);

      useDataSourceStore.setState({ isRealApi: true });
      await usePrintConfigStore.getState().setListingNumber(100);

      expect(isInitializing).toBe(true);
      expect(usePrintConfigStore.getState().isInitializing).toBe(false);
    });
  });

  describe('resetPrintParameters', () => {
    it('should reset printer parameters to VOID in mock mode', async () => {
      usePrintConfigStore.setState({
        currentConfig: MOCK_CONFIG,
      });

      await usePrintConfigStore.getState().resetPrintParameters();

      const state = usePrintConfigStore.getState();
      expect(state.currentConfig).toEqual({
        ...MOCK_CONFIG,
        currentPrinterName: 'VOID',
        currentPrinterNum: 0,
        numberCopies: 0,
      });
      expect(state.isInitializing).toBe(false);
      expect(state.error).toBeNull();
    });

    it('should use default config when currentConfig is null in mock mode', async () => {
      await usePrintConfigStore.getState().resetPrintParameters();

      expect(usePrintConfigStore.getState().currentConfig).toEqual(DEFAULT_CONFIG);
    });

    it('should call API and update config in real API mode', async () => {
      useDataSourceStore.setState({ isRealApi: true });

      vi.mocked(apiClient.post).mockResolvedValueOnce({
        data: { success: true, data: undefined },
      } as ApiResponse<void>);

      const resetConfig = { ...MOCK_CONFIG, currentPrinterName: 'VOID', currentPrinterNum: 0, numberCopies: 0 };
      vi.mocked(apiClient.get).mockResolvedValueOnce({
        data: { success: true, data: resetConfig },
      } as ApiResponse<PrintConfig>);

      await usePrintConfigStore.getState().resetPrintParameters();

      expect(apiClient.post).toHaveBeenCalledWith('/api/print-config/reset', {});
      expect(apiClient.get).toHaveBeenCalledWith('/api/print-config');
      expect(usePrintConfigStore.getState().currentConfig).toEqual(resetConfig);
    });

    it('should set error and throw on API failure', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      const errorMessage = 'Reset failed';

      vi.mocked(apiClient.post).mockRejectedValueOnce(new Error(errorMessage));

      await expect(
        usePrintConfigStore.getState().resetPrintParameters()
      ).rejects.toThrow(errorMessage);

      expect(usePrintConfigStore.getState().error).toBe(errorMessage);
    });
  });

  describe('getPrintConfig', () => {
    it('should return current config in mock mode', async () => {
      usePrintConfigStore.setState({ currentConfig: MOCK_CONFIG });

      const config = await usePrintConfigStore.getState().getPrintConfig();

      expect(config).toEqual(MOCK_CONFIG);
      expect(usePrintConfigStore.getState().isInitializing).toBe(false);
    });

    it('should return default config when currentConfig is null in mock mode', async () => {
      const config = await usePrintConfigStore.getState().getPrintConfig();

      expect(config).toEqual(DEFAULT_CONFIG);
    });

    it('should fetch config from API in real API mode', async () => {
      useDataSourceStore.setState({ isRealApi: true });

      vi.mocked(apiClient.get).mockResolvedValueOnce({
        data: { success: true, data: MOCK_CONFIG },
      } as ApiResponse<PrintConfig>);

      const config = await usePrintConfigStore.getState().getPrintConfig();

      expect(apiClient.get).toHaveBeenCalledWith('/api/print-config');
      expect(config).toEqual(MOCK_CONFIG);
      expect(usePrintConfigStore.getState().currentConfig).toEqual(MOCK_CONFIG);
    });

    it('should use default config when API returns null', async () => {
      useDataSourceStore.setState({ isRealApi: true });

      vi.mocked(apiClient.get).mockResolvedValueOnce({
        data: { success: true, data: null },
      } as ApiResponse<PrintConfig>);

      const config = await usePrintConfigStore.getState().getPrintConfig();

      expect(config).toEqual(DEFAULT_CONFIG);
      expect(usePrintConfigStore.getState().currentConfig).toEqual(DEFAULT_CONFIG);
    });

    it('should set error and throw on API failure', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      const errorMessage = 'Failed to fetch config';

      vi.mocked(apiClient.get).mockRejectedValueOnce(new Error(errorMessage));

      await expect(
        usePrintConfigStore.getState().getPrintConfig()
      ).rejects.toThrow(errorMessage);

      expect(usePrintConfigStore.getState().error).toBe(errorMessage);
    });
  });

  describe('setError', () => {
    it('should set error message', () => {
      const errorMessage = 'Custom error';

      usePrintConfigStore.getState().setError(errorMessage);

      expect(usePrintConfigStore.getState().error).toBe(errorMessage);
    });

    it('should clear error when null is passed', () => {
      usePrintConfigStore.setState({ error: 'Some error' });

      usePrintConfigStore.getState().setError(null);

      expect(usePrintConfigStore.getState().error).toBeNull();
    });
  });

  describe('reset', () => {
    it('should reset store to initial state', () => {
      usePrintConfigStore.setState({
        currentConfig: MOCK_CONFIG,
        isInitializing: true,
        error: 'Some error',
      });

      usePrintConfigStore.getState().reset();

      const state = usePrintConfigStore.getState();
      expect(state.currentConfig).toBeNull();
      expect(state.isInitializing).toBe(false);
      expect(state.error).toBeNull();
    });
  });
});
```
