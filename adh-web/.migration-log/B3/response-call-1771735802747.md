```tsx
import { useState } from 'react';
import { Button, Input } from '@/components/ui';
import { cn } from '@/lib/utils';

interface FiltersPanelProps {
  dateDebut: Date | null;
  dateFin: Date | null;
  numeroTel: string;
  onDateDebutChange: (date: Date | null) => void;
  onDateFinChange: (date: Date | null) => void;
  onNumeroTelChange: (value: string) => void;
  onRechercher: () => void;
  className?: string;
}

export const FiltersPanel = ({
  dateDebut,
  dateFin,
  numeroTel,
  onDateDebutChange,
  onDateFinChange,
  onNumeroTelChange,
  onRechercher,
  className,
}: FiltersPanelProps) => {
  const [dateDebutStr, setDateDebutStr] = useState(
    dateDebut ? dateDebut.toISOString().split('T')[0] : ''
  );
  const [dateFinStr, setDateFinStr] = useState(
    dateFin ? dateFin.toISOString().split('T')[0] : ''
  );

  const handleDateDebutChange = (value: string) => {
    setDateDebutStr(value);
    if (value) {
      const date = new Date(value);
      if (!isNaN(date.getTime())) {
        onDateDebutChange(date);
      }
    } else {
      onDateDebutChange(null);
    }
  };

  const handleDateFinChange = (value: string) => {
    setDateFinStr(value);
    if (value) {
      const date = new Date(value);
      if (!isNaN(date.getTime())) {
        onDateFinChange(date);
      }
    } else {
      onDateFinChange(null);
    }
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter') {
      onRechercher();
    }
  };

  return (
    <div className={cn('flex flex-col gap-4 p-4 bg-white border border-gray-300 rounded', className)}>
      <div className="flex gap-4 items-end flex-wrap">
        <div className="flex flex-col gap-2 min-w-[200px]">
          <label htmlFor="dateDebut" className="text-sm font-medium text-gray-700">
            Date début
          </label>
          <Input
            id="dateDebut"
            type="date"
            value={dateDebutStr}
            onChange={(e) => handleDateDebutChange(e.target.value)}
            onKeyDown={handleKeyDown}
            className="w-full"
          />
        </div>

        <div className="flex flex-col gap-2 min-w-[200px]">
          <label htmlFor="dateFin" className="text-sm font-medium text-gray-700">
            Date fin
          </label>
          <Input
            id="dateFin"
            type="date"
            value={dateFinStr}
            onChange={(e) => handleDateFinChange(e.target.value)}
            onKeyDown={handleKeyDown}
            className="w-full"
          />
        </div>

        <div className="flex flex-col gap-2 min-w-[200px]">
          <label htmlFor="numeroTel" className="text-sm font-medium text-gray-700">
            Numéro de téléphone
          </label>
          <Input
            id="numeroTel"
            type="text"
            value={numeroTel}
            onChange={(e) => onNumeroTelChange(e.target.value)}
            onKeyDown={handleKeyDown}
            placeholder="Numéro..."
            className="w-full"
          />
        </div>

        <Button
          onClick={onRechercher}
          className="px-6 py-2"
        >
          Rechercher
        </Button>
      </div>
    </div>
  );
};
```
