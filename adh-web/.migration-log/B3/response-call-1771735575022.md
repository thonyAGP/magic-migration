```typescript
import { create } from 'zustand';

import { apiClient } from '@/services/api/apiClient';

interface FacturationAppelState {
  historiqueAppels: HistoriqueAppel[];
  coefficients: CoefTelephone[];
  reseauStatus: ReseauCloture | null;
  isLoading: boolean;
  error: string | null;
  selectedAppel: HistoriqueAppel | null;
  filters: FacturationFilters;
}

interface FacturationAppelActions {
  chargerHistoriqueAppels: (societe: string, compte: number, filiation: number) => Promise<void>;
  recupererCoefficient: (qualite: string) => Promise<number>;
  facturer: (appel: HistoriqueAppel) => Promise<void>;
  debloquerCloture: () => Promise<void>;
  testerReseau: () => Promise<string>;
  annulerFacturation: (appelId: number) => Promise<void>;
  setSelectedAppel: (appel: HistoriqueAppel | null) => void;
  setFilters: (filters: FacturationFilters) => void;
  setError: (error: string | null) => void;
  reset: () => void;
}

type FacturationAppelStore = FacturationAppelState & FacturationAppelActions;

const MOCK_RESEAU_STATUS: ReseauCloture = {
  testReseau: 'OK',
  clotureEnCours: false,
};

const initialState: FacturationAppelState = {
  historiqueAppels: [],
  coefficients: [],
  reseauStatus: null,
  isLoading: false,
  error: null,
  selectedAppel: null,
  filters: {},
};

export const useFacturationAppelStore = create<FacturationAppelStore>()((set, get) => ({
  ...initialState,

  chargerHistoriqueAppels: async (societe, compte, filiation) => {
    const { isRealApi } = useDataSourceStore.getState();
    const { filters } = get();
    set({ isLoading: true, error: null });

    if (!isRealApi) {
      let filtered = [...MOCK_APPELS];
      
      if (filters.dateDebut) {
        filtered = filtered.filter((a) => a.dateAppel >= filters.dateDebut!);
      }
      if (filters.dateFin) {
        filtered = filtered.filter((a) => a.dateAppel <= filters.dateFin!);
      }
      if (filters.numeroTel) {
        filtered = filtered.filter((a) =>
          a.numeroTel.includes(filters.numeroTel!),
        );
      }

      filtered.sort((a, b) => {
        const dateCompare = b.dateAppel.getTime() - a.dateAppel.getTime();
        if (dateCompare !== 0) return dateCompare;
        return b.heureAppel.localeCompare(a.heureAppel);
      });

      set({ historiqueAppels: filtered, isLoading: false });
      return;
    }

    try {
      const response = await apiClient.get<GetHistoriqueAppelsResponse>(
        '/api/telephone/historique',
        {
          params: {
            societe,
            compte,
            filiation,
            dateDebut: filters.dateDebut?.toISOString().split('T')[0],
            dateFin: filters.dateFin?.toISOString().split('T')[0],
            numeroTel: filters.numeroTel,
          },
        },
      );
      set({ historiqueAppels: response.data.data ?? [] });
    } catch (e: unknown) {
      const message =
        e instanceof Error ? e.message : 'Erreur chargement historique appels';
      set({ historiqueAppels: [], error: message });
    } finally {
      set({ isLoading: false });
    }
  },

  recupererCoefficient: async (qualite) => {
    const { isRealApi } = useDataSourceStore.getState();
    set({ isLoading: true, error: null });

    if (!isRealApi) {
      const coef = MOCK_COEFFICIENTS.find((c) => c.qualite === qualite);
      set({ isLoading: false });
      return coef?.coefficient ?? 1.0;
    }

    try {
      const response = await apiClient.get<GetCoefficientResponse>(
        `/api/telephone/coefficient/${qualite}`,
      );
      const coefficient = response.data.data?.coefficient ?? 1.0;
      
      set((state) => ({
        coefficients: [
          ...state.coefficients.filter((c) => c.qualite !== qualite),
          { qualite, coefficient },
        ],
      }));
      
      return coefficient;
    } catch (e: unknown) {
      const message =
        e instanceof Error ? e.message : 'Erreur récupération coefficient';
      set({ error: message });
      return 1.0;
    } finally {
      set({ isLoading: false });
    }
  },

  facturer: async (appel) => {
    const { isRealApi } = useDataSourceStore.getState();
    const { reseauStatus } = get();
    set({ isLoading: true, error: null });

    if (reseauStatus?.clotureEnCours) {
      set({
        error: 'Impossible de facturer : clôture en cours',
        isLoading: false,
      });
      return;
    }

    if (!isRealApi) {
      set({ isLoading: false });
      return;
    }

    try {
      await apiClient.post<FacturerResponse>('/api/telephone/facturer', {
        societe: appel.societe,
        appel,
      });
      
      set((state) => ({
        historiqueAppels: state.historiqueAppels.map((a) =>
          a === appel ? { ...a, montant: appel.montant } : a,
        ),
      }));
    } catch (e: unknown) {
      const message = e instanceof Error ? e.message : 'Erreur facturation';
      set({ error: message });
    } finally {
      set({ isLoading: false });
    }
  },

  debloquerCloture: async () => {
    const { isRealApi } = useDataSourceStore.getState();
    set({ isLoading: true, error: null });

    if (!isRealApi) {
      set({
        reseauStatus: { ...MOCK_RESEAU_STATUS, clotureEnCours: false },
        isLoading: false,
      });
      return;
    }

    try {
      await apiClient.post<DebloquerClotureResponse>('/api/reseau/debloquer');
      
      set((state) => ({
        reseauStatus: state.reseauStatus
          ? { ...state.reseauStatus, clotureEnCours: false }
          : null,
      }));
    } catch (e: unknown) {
      const message =
        e instanceof Error ? e.message : 'Erreur déblocage clôture';
      set({ error: message });
    } finally {
      set({ isLoading: false });
    }
  },

  testerReseau: async () => {
    const { isRealApi } = useDataSourceStore.getState();
    set({ isLoading: true, error: null });

    if (!isRealApi) {
      set({ reseauStatus: MOCK_RESEAU_STATUS, isLoading: false });
      return 'OK';
    }

    try {
      const response = await apiClient.get<TesterReseauResponse>(
        '/api/reseau/test',
      );
      const status = response.data.data?.status ?? 'ERROR';
      
      const reseauResponse = await apiClient.get<GetReseauStatusResponse>(
        '/api/reseau/status',
      );
      set({ reseauStatus: reseauResponse.data.data ?? null });
      
      return status;
    } catch (e: unknown) {
      const message = e instanceof Error ? e.message : 'Erreur test réseau';
      set({ error: message });
      return 'ERROR';
    } finally {
      set({ isLoading: false });
    }
  },

  annulerFacturation: async (appelId) => {
    const { isRealApi } = useDataSourceStore.getState();
    set({ isLoading: true, error: null });

    if (!isRealApi) {
      set({ isLoading: false });
      return;
    }

    try {
      await apiClient.post<AnnulerFacturationResponse>(
        `/api/telephone/annuler/${appelId}`,
      );
      
      set((state) => ({
        historiqueAppels: state.historiqueAppels.filter(
          (a) => (a as unknown as { id?: number }).id !== appelId,
        ),
      }));
    } catch (e: unknown) {
      const message =
        e instanceof Error ? e.message : 'Erreur annulation facturation';
      set({ error: message });
    } finally {
      set({ isLoading: false });
    }
  },

  setSelectedAppel: (appel) => {
    set({ selectedAppel: appel, error: null });
  },

  setFilters: (filters) => {
    set({ filters, error: null });
  },

  setError: (error) => {
    set({ error });
  },

  reset: () => set({ ...initialState }),
}));
```
