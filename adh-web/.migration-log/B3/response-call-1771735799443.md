```tsx
import { useState } from 'react';
import { Button, Input } from '@/components/ui';
import { cn } from '@/lib/utils';

interface FiltersPanelProps {
  dateDebut: Date | null;
  dateFin: Date | null;
  numeroTel: string;
  onDateDebutChange: (date: Date | null) => void;
  onDateFinChange: (date: Date | null) => void;
  onNumeroTelChange: (numeroTel: string) => void;
  onSearch: () => void;
  onReset?: () => void;
  className?: string;
}

export const FiltersPanel = ({
  dateDebut,
  dateFin,
  numeroTel,
  onDateDebutChange,
  onDateFinChange,
  onNumeroTelChange,
  onSearch,
  onReset,
  className,
}: FiltersPanelProps) => {
  const [localDateDebut, setLocalDateDebut] = useState<string>(
    dateDebut ? dateDebut.toISOString().split('T')[0] : ''
  );
  const [localDateFin, setLocalDateFin] = useState<string>(
    dateFin ? dateFin.toISOString().split('T')[0] : ''
  );
  const [localNumeroTel, setLocalNumeroTel] = useState<string>(numeroTel);

  const handleDateDebutChange = (value: string) => {
    setLocalDateDebut(value);
    onDateDebutChange(value ? new Date(value) : null);
  };

  const handleDateFinChange = (value: string) => {
    setLocalDateFin(value);
    onDateFinChange(value ? new Date(value) : null);
  };

  const handleNumeroTelChange = (value: string) => {
    setLocalNumeroTel(value);
    onNumeroTelChange(value);
  };

  const handleReset = () => {
    setLocalDateDebut('');
    setLocalDateFin('');
    setLocalNumeroTel('');
    onDateDebutChange(null);
    onDateFinChange(null);
    onNumeroTelChange('');
    onReset?.();
  };

  return (
    <div className={cn('bg-white rounded-lg shadow-sm border border-gray-200 p-4', className)}>
      <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mb-4">
        <div>
          <label htmlFor="dateDebut" className="block text-sm font-medium text-gray-700 mb-1">
            Date début
          </label>
          <Input
            id="dateDebut"
            type="date"
            value={localDateDebut}
            onChange={(e) => handleDateDebutChange(e.target.value)}
            className="w-full"
          />
        </div>

        <div>
          <label htmlFor="dateFin" className="block text-sm font-medium text-gray-700 mb-1">
            Date fin
          </label>
          <Input
            id="dateFin"
            type="date"
            value={localDateFin}
            onChange={(e) => handleDateFinChange(e.target.value)}
            className="w-full"
          />
        </div>

        <div>
          <label htmlFor="numeroTel" className="block text-sm font-medium text-gray-700 mb-1">
            Numéro téléphone
          </label>
          <Input
            id="numeroTel"
            type="text"
            value={localNumeroTel}
            onChange={(e) => handleNumeroTelChange(e.target.value)}
            placeholder="Entrez le numéro..."
            className="w-full"
          />
        </div>
      </div>

      <div className="flex gap-2">
        <Button onClick={onSearch} className="flex-1 md:flex-none">
          Rechercher
        </Button>
        {onReset && (
          <Button onClick={handleReset} variant="outline" className="flex-1 md:flex-none">
            Réinitialiser
          </Button>
        )}
      </div>
    </div>
  );
};
```
