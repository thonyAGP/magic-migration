```tsx
import { useState, useEffect, useCallback } from 'react';
import { useNavigate } from 'react-router-dom';
import { ScreenLayout } from '@/components/layout';
import { Button } from '@/components/ui';
import { useFacturationAppelStore } from '@/stores/facturationAppelStore';
import { useAuthStore } from '@/stores';
import type { HistoriqueAppel, TypeCompte } from '@/types/facturationAppel';
import { cn } from '@/lib/utils';

export const FacturationAppelPage = () => {
  const navigate = useNavigate();
  const user = useAuthStore((s) => s.user);

  const historiqueAppels = useFacturationAppelStore((s) => s.historiqueAppels);
  const coefficientTelephone = useFacturationAppelStore((s) => s.coefficientTelephone);
  const cloture = useFacturationAppelStore((s) => s.cloture);
  const isLoading = useFacturationAppelStore((s) => s.isLoading);
  const error = useFacturationAppelStore((s) => s.error);
  const filterSociete = useFacturationAppelStore((s) => s.filterSociete);
  const filterDateDebut = useFacturationAppelStore((s) => s.filterDateDebut);
  const filterDateFin = useFacturationAppelStore((s) => s.filterDateFin);

  const chargerHistoriqueAppels = useFacturationAppelStore((s) => s.chargerHistoriqueAppels);
  const recupererCoefficient = useFacturationAppelStore((s) => s.recupererCoefficient);
  const facturerAppel = useFacturationAppelStore((s) => s.facturerAppel);
  const verifierCloture = useFacturationAppelStore((s) => s.verifierCloture);
  const debloquerCloture = useFacturationAppelStore((s) => s.debloquerCloture);
  const marquerGratuit = useFacturationAppelStore((s) => s.marquerGratuit);
  const annulerFacturation = useFacturationAppelStore((s) => s.annulerFacturation);
  const setFilterSociete = useFacturationAppelStore((s) => s.setFilterSociete);
  const setFilterDateDebut = useFacturationAppelStore((s) => s.setFilterDateDebut);
  const setFilterDateFin = useFacturationAppelStore((s) => s.setFilterDateFin);
  const resetFilters = useFacturationAppelStore((s) => s.resetFilters);
  const setError = useFacturationAppelStore((s) => s.setError);
  const reset = useFacturationAppelStore((s) => s.reset);

  const [selectedAppel, setSelectedAppel] = useState<HistoriqueAppel | null>(null);
  const [typeCompte, setTypeCompte] = useState<TypeCompte>('GO');
  const [numeroCompte, setNumeroCompte] = useState('');
  const [filiation, setFiliation] = useState('');
  const [appelGratuit, setAppelGratuit] = useState(false);
  const [raisonGratuite, setRaisonGratuite] = useState('');
  const [showFacturationPanel, setShowFacturationPanel] = useState(false);

  const societe = 'ADH';
  const prefixe = user?.prefixe || 'CAI01';

  useEffect(() => {
    recupererCoefficient();
    verifierCloture();
    return () => reset();
  }, [recupererCoefficient, verifierCloture, reset]);

  const handleRechercher = useCallback(() => {
    if (!filterDateDebut || !filterDateFin) {
      setError('Veuillez sélectionner une période de dates');
      return;
    }
    chargerHistoriqueAppels(societe, prefixe, filterDateDebut, filterDateFin);
  }, [chargerHistoriqueAppels, societe, prefixe, filterDateDebut, filterDateFin, setError]);

  const handleFacturerClick = useCallback((appel: HistoriqueAppel) => {
    setSelectedAppel(appel);
    setShowFacturationPanel(true);
    setAppelGratuit(false);
    setRaisonGratuite('');
    setNumeroCompte('');
    setFiliation('');
    setTypeCompte('GO');
  }, []);

  const handleMarquerGratuitClick = useCallback(
    async (appel: HistoriqueAppel) => {
      const raison = prompt('Raison de la gratuité:');
      if (!raison) return;
      if (!appel.id) return;
      await marquerGratuit(appel.id, raison);
      handleRechercher();
    },
    [marquerGratuit, handleRechercher],
  );

  const handleAnnulerFacturation = useCallback(
    async (appelId: number) => {
      await annulerFacturation(appelId);
      handleRechercher();
    },
    [annulerFacturation, handleRechercher],
  );

  const handleConfirmerFacturation = useCallback(async () => {
    if (!selectedAppel) return;

    if (appelGratuit && !raisonGratuite.trim()) {
      setError('Veuillez indiquer une raison pour la gratuité');
      return;
    }

    if (!appelGratuit && (!numeroCompte || !filiation)) {
      setError('Veuillez renseigner le numéro de compte et la filiation');
      return;
    }

    if (appelGratuit && selectedAppel.id) {
      await marquerGratuit(selectedAppel.id, raisonGratuite);
    } else {
      await facturerAppel({
        appel: selectedAppel,
        numeroCompte: parseInt(numeroCompte, 10),
        filiation: parseInt(filiation, 10),
        typeCompte,
      });
    }

    setShowFacturationPanel(false);
    setSelectedAppel(null);
    handleRechercher();
  }, [
    selectedAppel,
    appelGratuit,
    raisonGratuite,
    numeroCompte,
    filiation,
    typeCompte,
    facturerAppel,
    marquerGratuit,
    setError,
    handleRechercher,
  ]);

  const handleDebloquerCloture = useCallback(async () => {
    await debloquerCloture();
    verifierCloture();
  }, [debloquerCloture, verifierCloture]);

  const montantCalcule = selectedAppel
    ? (selectedAppel.montant * (coefficientTelephone || 1)).toFixed(2)
    : '0.00';

  return (
    <ScreenLayout>
      <div className="space-y-6 max-w-7xl mx-auto">
        <div className="flex items-center justify-between">
          <div>
            <h2 className="text-xl font-semibold">Facturation Appels Téléphoniques</h2>
            <p className="text-on-surface-muted text-sm mt-1">
              Gérer la facturation des appels téléphoniques
            </p>
          </div>
          {user && (
            <span className="text-xs text-on-surface-muted">
              {user.prenom} {user.nom} - {prefixe}
            </span>
          )}
        </div>

        {error && (
          <div className="bg-red-50 border border-red-200 text-red-700 px-4 py-3 rounded-md text-sm">
            {error}
          </div>
        )}

        <div className="bg-surface border border-border rounded-lg p-4 space-y-4">
          <h3 className="font-medium">Filtres de recherche</h3>
          <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
            <div>
              <label className="block text-sm font-medium mb-1">Date début</label>
              <input
                type="date"
                value={filterDateDebut?.toISOString().split('T')[0] || ''}
                onChange={(e) => setFilterDateDebut(e.target.value ? new Date(e.target.value) : null)}
                className="w-full px-3 py-2 border border-border rounded-md"
              />
            </div>
            <div>
              <label className="block text-sm font-medium mb-1">Date fin</label>
              <input
                type="date"
                value={filterDateFin?.toISOString().split('T')[0] || ''}
                onChange={(e) => setFilterDateFin(e.target.value ? new Date(e.target.value) : null)}
                className="w-full px-3 py-2 border border-border rounded-md"
              />
            </div>
            <div className="flex items-end">
              <Button onClick={handleRechercher} disabled={isLoading} className="w-full">
                {isLoading ? 'Recherche...' : 'Rechercher'}
              </Button>
            </div>
          </div>
        </div>

        <div className="bg-surface border border-border rounded-lg overflow-hidden">
          <div className="overflow-x-auto">
            <table className="w-full">
              <thead className="bg-surface-muted border-b border-border">
                <tr>
                  <th className="px-4 py-3 text-left text-sm font-medium">Date</th>
                  <th className="px-4 py-3 text-left text-sm font-medium">Heure</th>
                  <th className="px-4 py-3 text-left text-sm font-medium">Numéro</th>
                  <th className="px-4 py-3 text-right text-sm font-medium">Durée</th>
                  <th className="px-4 py-3 text-right text-sm font-medium">Montant</th>
                  <th className="px-4 py-3 text-left text-sm font-medium">Qualité</th>
                  <th className="px-4 py-3 text-left text-sm font-medium">Statut</th>
                  <th className="px-4 py-3 text-left text-sm font-medium">Actions</th>
                </tr>
              </thead>
              <tbody className="divide-y divide-border">
                {historiqueAppels.length === 0 && !isLoading && (
                  <tr>
                    <td colSpan={8} className="px-4 py-8 text-center text-on-surface-muted">
                      Aucun appel trouvé
                    </td>
                  </tr>
                )}
                {historiqueAppels.map((appel, idx) => (
                  <tr key={appel.id || idx} className="hover:bg-surface-hover">
                    <td className="px-4 py-3 text-sm">
                      {new Date(appel.dateAppel).toLocaleDateString('fr-FR')}
                    </td>
                    <td className="px-4 py-3 text-sm">{appel.heureAppel}</td>
                    <td className="px-4 py-3 text-sm font-mono">{appel.numeroTel}</td>
                    <td className="px-4 py-3 text-sm text-right">{appel.duree}</td>
                    <td className="px-4 py-3 text-sm text-right">{appel.montant.toFixed(2)} €</td>
                    <td className="px-4 py-3 text-sm">
                      <span
                        className={cn(
                          'px-2 py-1 rounded text-xs font-medium',
                          appel.qualite === 'OK' && 'bg-green-100 text-green-800',
                          appel.qualite === 'Mauvaise' && 'bg-red-100 text-red-800',
                          appel.qualite === 'Interruption' && 'bg-yellow-100 text-yellow-800',
                        )}
                      >
                        {appel.qualite || 'N/A'}
                      </span>
                    </td>
                    <td className="px-4 py-3 text-sm">
                      {appel.gratuite && (
                        <span className="px-2 py-1 bg-blue-100 text-blue-800 rounded text-xs font-medium">
                          Gratuit
                        </span>
                      )}
                      {appel.facture && (
                        <span className="px-2 py-1 bg-purple-100 text-purple-800 rounded text-xs font-medium">
                          Facturé
                        </span>
                      )}
                      {!appel.gratuite && !appel.facture && (
                        <span className="px-2 py-1 bg-gray-100 text-gray-800 rounded text-xs font-medium">
                          En attente
                        </span>
                      )}
                    </td>
                    <td className="px-4 py-3 text-sm">
                      <div className="flex gap-2">
                        {!appel.facture && !appel.gratuite && (
                          <>
                            <button
                              onClick={() => handleFacturerClick(appel)}
                              disabled={cloture?.cloture_enCours || false}
                              className="px-3 py-1 bg-primary text-white rounded text-xs hover:bg-primary-dark disabled:opacity-50"
                            >
                              Facturer
                            </button>
                            <button
                              onClick={() => handleMarquerGratuitClick(appel)}
                              className="px-3 py-1 border border-border rounded text-xs hover:bg-surface-hover"
                            >
                              Gratuit
                            </button>
                          </>
                        )}
                        {appel.facture && appel.id && (
                          <button
                            onClick={() => handleAnnulerFacturation(appel.id!)}
                            className="px-3 py-1 bg-red-100 text-red-800 rounded text-xs hover:bg-red-200"
                          >
                            Annuler
                          </button>
                        )}
                      </div>
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          </div>
        </div>

        {showFacturationPanel && selectedAppel && (
          <div className="fixed inset-0 bg-black/50 flex items-center justify-center z-50">
            <div className="bg-surface border border-border rounded-lg p-6 w-full max-w-md space-y-4">
              <h3 className="text-lg font-semibold">Facturation Appel</h3>

              <div className="space-y-3">
                <div>
                  <label className="flex items-center gap-2">
                    <input
                      type="checkbox"
                      checked={appelGratuit}
                      onChange={(e) => setAppelGratuit(e.target.checked)}
                      className="rounded"
                    />
                    <span className="text-sm font-medium">Appel gratuit</span>
                  </label>
                </div>

                {appelGratuit ? (
                  <div>
                    <label className="block text-sm font-medium mb-1">Raison gratuité</label>
                    <input
                      type="text"
                      value={raisonGratuite}
                      onChange={(e) => setRaisonGratuite(e.target.value)}
                      className="w-full px-3 py-2 border border-border rounded-md"
                      placeholder="Raison..."
                    />
                  </div>
                ) : (
                  <>
                    <div>
                      <label className="block text-sm font-medium mb-1">Type compte</label>
                      <div className="flex gap-4">
                        <label className="flex items-center gap-2">
                          <input
                            type="radio"
                            checked={typeCompte === 'GO'}
                            onChange={() => setTypeCompte('GO')}
                          />
                          <span className="text-sm">GO</span>
                        </label>
                        <label className="flex items-center gap-2">
                          <input
                            type="radio"
                            checked={typeCompte === 'GM'}
                            onChange={() => setTypeCompte('GM')}
                          />
                          <span className="text-sm">GM</span>
                        </label>
                      </div>
                    </div>

                    <div>
                      <label className="block text-sm font-medium mb-1">Numéro compte</label>
                      <input
                        type="text"
                        value={numeroCompte}
                        onChange={(e) => setNumeroCompte(e.target.value)}
                        className="w-full px-3 py-2 border border-border rounded-md"
                      />
                    </div>

                    <div>
                      <label className="block text-sm font-medium mb-1">Filiation</label>
                      <input
                        type="text"
                        value={filiation}
                        onChange={(e) => setFiliation(e.target.value)}
                        className="w-full px-3 py-2 border border-border rounded-md"
                      />
                    </div>

                    <div>
                      <label className="block text-sm font-medium mb-1">Montant calculé</label>
                      <input
                        type="text"
                        value={`${montantCalcule} €`}
                        readOnly
                        className="w-full px-3 py-2 border border-border rounded-md bg-surface-muted"
                      />
                    </div>
                  </>
                )}
              </div>

              <div className="flex gap-3 justify-end">
                <button
                  onClick={() => {
                    setShowFacturationPanel(false);
                    setSelectedAppel(null);
                  }}
                  className="px-4 py-2 border border-border rounded-md hover:bg-surface-hover"
                >
                  Annuler
                </button>
                <Button onClick={handleConfirmerFacturation}>Confirmer</Button>
              </div>
            </div>
          </div>
        )}

        <div className="bg-surface border border-border rounded-lg p-4">
          <div className="flex items-center justify-between">
            <div className="space-y-1">
              <div className="text-sm">
                <span className="font-medium">Coefficient téléphone:</span>{' '}
                <span className="text-on-surface-muted">
                  {coefficientTelephone !== null ? coefficientTelephone : 'N/A'}
                </span>
              </div>
              <div className="text-sm">
                <span className="font-medium">Statut clôture:</span>{' '}
                <span
                  className={cn(
                    'font-medium',
                    cloture?.cloture_enCours ? 'text-red-600' : 'text-green-600',
                  )}
                >
                  {cloture?.cloture_enCours ? 'Clôture en cours' : 'Ouvert'}
                </span>
              </div>
            </div>
            {cloture?.cloture_enCours && (
              <Button onClick={handleDebloquerCloture} variant="outline">
                Débloquer clôture
              </Button>
            )}
          </div>
        </div>

        <div className="flex justify-start">
          <button
            onClick={() => navigate('/caisse/menu')}
            className="px-4 py-2 border border-border rounded-md text-on-surface hover:bg-surface-hover"
          >
            Retour au menu
          </button>
        </div>
      </div>
    </ScreenLayout>
  );
};
```
