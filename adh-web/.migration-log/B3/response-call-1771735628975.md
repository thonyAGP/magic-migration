```typescript
import { apiClient, type ApiResponse } from "@/services/api/apiClient";
import type {
  HistoriqueAppel,
  CoefTelephone,
  ReseauCloture,
  FacturationRequest,
  FacturationResponse,
  MarquerGratuitRequest,
  GetHistoriqueAppelsResponse,
  GetCoefficientResponse,
  FacturerAppelResponse,
  GetClotureStatusResponse,
  DebloquerClotureResponse,
  MarquerGratuitResponse,
} from "@/types/facturationAppel";

export const facturationAppelApi = {
  getHistoriqueAppels: (
    societe: string,
    dateDebut?: string,
    dateFin?: string,
    numeroTel?: string,
  ) => {
    const params = new URLSearchParams({
      societe,
      ...(dateDebut && { dateDebut }),
      ...(dateFin && { dateFin }),
      ...(numeroTel && { numeroTel }),
    });

    return apiClient.get<GetHistoriqueAppelsResponse>(
      `/api/facturation-appel/historiques?${params.toString()}`,
    );
  },

  getCoefficient: (societe: string) =>
    apiClient.get<GetCoefficientResponse>(
      `/api/facturation-appel/coefficient/${societe}`,
    ),

  getClotureStatus: (societe: string) =>
    apiClient.get<GetClotureStatusResponse>(
      `/api/facturation-appel/cloture-status/${societe}`,
    ),

  debloquerCloture: (societe: string) =>
    apiClient.post<DebloquerClotureResponse>(
      `/api/facturation-appel/debloquer-cloture?societe=${societe}`,
      {},
    ),

  creerFacture: (data: FacturationRequest) =>
    apiClient.post<FacturerAppelResponse>(
      "/api/facturation-appel/creer-facture",
      data,
    ),

  marquerGratuit: (appelId: number, data: MarquerGratuitRequest) =>
    apiClient.patch<MarquerGratuitResponse>(
      `/api/facturation-appel/marquer-gratuit/${appelId}`,
      data,
    ),
};
```
