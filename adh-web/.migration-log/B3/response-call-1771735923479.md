```tsx
import { Button } from "@/components/ui";
import { cn } from "@/lib/utils";
import { useDataSourceStore } from "@/stores/dataSourceStore";

interface ActionsPanelProps {
  className?: string;
  onFacturer: () => void;
  onAnnuler: () => void;
  isFacturerDisabled?: boolean;
  isLoading?: boolean;
}

export const ActionsPanel = ({
  className,
  onFacturer,
  onAnnuler,
  isFacturerDisabled = false,
  isLoading = false,
}: ActionsPanelProps) => {
  const isRealApi = useDataSourceStore((s) => s.isRealApi);

  return (
    <div className={cn("flex items-center justify-between gap-4 p-4 border-t bg-gray-50", className)}>
      <div className="flex items-center gap-2">
        <div
          className={cn(
            "w-3 h-3 rounded-full",
            isRealApi ? "bg-green-500" : "bg-yellow-500"
          )}
          title={isRealApi ? "API réelle connectée" : "Mode démonstration (mock)"}
        />
        <span className="text-sm text-gray-600">
          {isRealApi ? "Réseau: Connecté" : "Mode: Démo"}
        </span>
      </div>

      <div className="flex items-center gap-3">
        <Button
          variant="danger"
          onClick={onAnnuler}
          disabled={isLoading}
        >
          Annuler
        </Button>
        <Button
          variant="primary"
          onClick={onFacturer}
          disabled={isFacturerDisabled || isLoading}
        >
          {isLoading ? "Facturation..." : "Facturer"}
        </Button>
      </div>
    </div>
  );
};
```
