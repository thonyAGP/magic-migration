```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { useFacturationAppelStore } from '@/stores/facturationAppelStore';
import { useDataSourceStore } from '@/stores/dataSourceStore';
import { apiClient } from '@/services/api/apiClient';
import type {
  HistoriqueAppel,
  GetHistoriqueAppelsResponse,
  GetCoefficientResponse,
  FacturerAppelResponse,
  GetClotureStatusResponse,
  DebloquerClotureResponse,
  MarquerGratuitResponse,
  AnnulerFacturationResponse,
} from '@/types/facturationAppel';

vi.mock('@/services/api/apiClient', () => ({
  apiClient: {
    get: vi.fn(),
    post: vi.fn(),
    put: vi.fn(),
    delete: vi.fn(),
  },
}));

  societe: 'SOC1',

const MOCK_APPEL_GRATUIT: HistoriqueAppel = {
  id: 4,
  societe: 'SOC1',
  prefixe: 'CAI02',
  dateAppel: new Date('2026-02-18T16:30:00'),
  heureAppel: '16:30:05',

describe('facturationAppelStore', () => {
  beforeEach(() => {
    useFacturationAppelStore.setState({
      historiqueAppels: [],
      coefficientTelephone: null,
      cloture: null,
      isLoading: false,
      error: null,
      filterSociete: '',
      filterDateDebut: null,
      filterDateFin: null,
    });
    useDataSourceStore.setState({ isRealApi: false });
    vi.clearAllMocks();
  });

  describe('chargerHistoriqueAppels', () => {
    it('should load mock data when isRealApi is false', async () => {
      const store = useFacturationAppelStore.getState();

      await store.chargerHistoriqueAppels('SOC1', 'CAI01');

      const state = useFacturationAppelStore.getState();
      expect(state.isLoading).toBe(false);
      expect(state.historiqueAppels.length).toBeGreaterThan(0);
      expect(state.historiqueAppels.every((a) => a.societe === 'SOC1')).toBe(true);
      expect(state.historiqueAppels.every((a) => a.prefixe === 'CAI01')).toBe(true);
      expect(state.error).toBe(null);
    });

    it('should filter by date range in mock mode', async () => {
      const store = useFacturationAppelStore.getState();
      const dateDebut = new Date('2026-02-19T00:00:00');
      const dateFin = new Date('2026-02-20T23:59:59');

      await store.chargerHistoriqueAppels('SOC1', 'CAI01', dateDebut, dateFin);

      const state = useFacturationAppelStore.getState();
      expect(state.historiqueAppels.every((a) => a.dateAppel >= dateDebut)).toBe(true);
      expect(state.historiqueAppels.every((a) => a.dateAppel <= dateFin)).toBe(true);
    });

    it('should sort by date and time descending', async () => {
      const store = useFacturationAppelStore.getState();

      await store.chargerHistoriqueAppels('SOC1', 'CAI01');

      const state = useFacturationAppelStore.getState();
      for (let i = 0; i < state.historiqueAppels.length - 1; i++) {
        const current = state.historiqueAppels[i];
        const next = state.historiqueAppels[i + 1];
        const currentTime = current.dateAppel.getTime();
        const nextTime = next.dateAppel.getTime();
        expect(currentTime).toBeGreaterThanOrEqual(nextTime);
      }
    });

    it('should load from API when isRealApi is true', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      const mockResponse: GetHistoriqueAppelsResponse = {
        success: true,
        data: [MOCK_APPEL_1, MOCK_APPEL_2],
      };
      vi.mocked(apiClient.get).mockResolvedValueOnce({ data: mockResponse });

      const store = useFacturationAppelStore.getState();
      await store.chargerHistoriqueAppels('SOC1', 'CAI01');

      const state = useFacturationAppelStore.getState();
      expect(state.isLoading).toBe(false);
      expect(state.historiqueAppels).toEqual([MOCK_APPEL_1, MOCK_APPEL_2]);
      expect(state.error).toBe(null);
      expect(apiClient.get).toHaveBeenCalledWith(
        expect.stringContaining('/api/facturation-appel/historique')
      );
    });

    it('should handle API error', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      const errorMessage = 'Network error';
      vi.mocked(apiClient.get).mockRejectedValueOnce(new Error(errorMessage));

      const store = useFacturationAppelStore.getState();
      await store.chargerHistoriqueAppels('SOC1', 'CAI01');

      const state = useFacturationAppelStore.getState();
      expect(state.isLoading).toBe(false);
      expect(state.historiqueAppels).toEqual([]);
      expect(state.error).toBe(errorMessage);
    });

    it('should set loading state during API call', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      const mockResponse: GetHistoriqueAppelsResponse = {
        success: true,
        data: [MOCK_APPEL_1],
      };
      
      let loadingDuringCall = false;
      vi.mocked(apiClient.get).mockImplementation(async () => {
        loadingDuringCall = useFacturationAppelStore.getState().isLoading;
        return { data: mockResponse };
      });

      const store = useFacturationAppelStore.getState();
      await store.chargerHistoriqueAppels('SOC1', 'CAI01');

      expect(loadingDuringCall).toBe(true);
      expect(useFacturationAppelStore.getState().isLoading).toBe(false);
    });
  });

  describe('recupererCoefficient', () => {
    it('should load mock coefficient when isRealApi is false', async () => {
      const store = useFacturationAppelStore.getState();

      await store.recupererCoefficient();

      const state = useFacturationAppelStore.getState();
      expect(state.coefficientTelephone).toBe(0.15);
      expect(state.isLoading).toBe(false);
      expect(state.error).toBe(null);
    });

    it('should load from API when isRealApi is true', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      const mockResponse: GetCoefficientResponse = {
        success: true,
        data: { coefficient: 0.18 },
      };
      vi.mocked(apiClient.get).mockResolvedValueOnce({ data: mockResponse });

      const store = useFacturationAppelStore.getState();
      await store.recupererCoefficient();

      const state = useFacturationAppelStore.getState();
      expect(state.coefficientTelephone).toBe(0.18);
      expect(state.error).toBe(null);
      expect(apiClient.get).toHaveBeenCalledWith('/api/facturation-appel/coefficient');
    });

    it('should handle API error', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      vi.mocked(apiClient.get).mockRejectedValueOnce(new Error('API error'));

      const store = useFacturationAppelStore.getState();
      await store.recupererCoefficient();

      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe('API error');
      expect(state.isLoading).toBe(false);
    });
  });

  describe('facturerAppel', () => {
    it('should reject facturation of gratuit appel', async () => {
      const store = useFacturationAppelStore.getState();

      await store.facturerAppel({
        appel: MOCK_APPEL_GRATUIT,
        numeroCompte: 12345,
        filiation: 1,
        typeCompte: 'GO',
      });

      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe('Appel gratuit - facturation non autorisée');
      expect(state.isLoading).toBe(false);
    });

    it('should block facturation when cloture is active', async () => {
      vi.mocked(apiClient.get).mockResolvedValueOnce({
        data: {
          success: true,
          data: { cloture_enCours: true, testReseau: 'ERREUR' },
        } as GetClotureStatusResponse,
      });

      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_1],
      });

      const store = useFacturationAppelStore.getState();
      await store.facturerAppel({
        appel: MOCK_APPEL_1,
        numeroCompte: 12345,
        filiation: 1,
        typeCompte: 'GO',
      });

      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe('Opération bloquée - clôture en cours');
    });

    it('should facturer appel in mock mode', async () => {
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_1],
        cloture: { cloture_enCours: false, testReseau: 'OK' },
      });

      const store = useFacturationAppelStore.getState();
      await store.facturerAppel({
        appel: MOCK_APPEL_1,
        numeroCompte: 12345,
        filiation: 1,
        typeCompte: 'GO',
      });

      const state = useFacturationAppelStore.getState();
      const updatedAppel = state.historiqueAppels.find((a) => a.id === 1);
      expect(updatedAppel?.facture).toBe(true);
      expect(updatedAppel?.operationId).toBeDefined();
      expect(state.error).toBe(null);
    });

    it('should facturer appel via API', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_1],
        cloture: { cloture_enCours: false, testReseau: 'OK' },
      });

      vi.mocked(apiClient.get).mockResolvedValueOnce({
        data: {
          success: true,
          data: { cloture_enCours: false, testReseau: 'OK' },
        } as GetClotureStatusResponse,
      });

      const mockResponse: FacturerAppelResponse = {
        success: true,
        data: { success: true, operationId: 50123 },
      };
      vi.mocked(apiClient.post).mockResolvedValueOnce({ data: mockResponse });

      const store = useFacturationAppelStore.getState();
      await store.facturerAppel({
        appel: MOCK_APPEL_1,
        numeroCompte: 12345,
        filiation: 1,
        typeCompte: 'GO',
      });

      const state = useFacturationAppelStore.getState();
      const updatedAppel = state.historiqueAppels.find((a) => a.id === 1);
      expect(updatedAppel?.facture).toBe(true);
      expect(updatedAppel?.operationId).toBe(50123);
      expect(apiClient.post).toHaveBeenCalledWith(
        '/api/facturation-appel/facturer',
        expect.any(Object)
      );
    });

    it('should handle API error during facturation', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_1],
        cloture: { cloture_enCours: false, testReseau: 'OK' },
      });

      vi.mocked(apiClient.get).mockResolvedValueOnce({
        data: {
          success: true,
          data: { cloture_enCours: false, testReseau: 'OK' },
        } as GetClotureStatusResponse,
      });

      vi.mocked(apiClient.post).mockRejectedValueOnce(new Error('Facturation failed'));

      const store = useFacturationAppelStore.getState();
      await store.facturerAppel({
        appel: MOCK_APPEL_1,
        numeroCompte: 12345,
        filiation: 1,
        typeCompte: 'GO',
      });

      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe('Facturation failed');
    });
  });

  describe('verifierCloture', () => {
    it('should return mock cloture status when isRealApi is false', async () => {
      const store = useFacturationAppelStore.getState();

      const result = await store.verifierCloture();

      const state = useFacturationAppelStore.getState();
      expect(result).toBe(false);
      expect(state.cloture?.cloture_enCours).toBe(false);
      expect(state.cloture?.testReseau).toBe('OK');
    });

    it('should return API cloture status when isRealApi is true', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      const mockResponse: GetClotureStatusResponse = {
        success: true,
        data: { cloture_enCours: true, testReseau: 'ERREUR' },
      };
      vi.mocked(apiClient.get).mockResolvedValueOnce({ data: mockResponse });

      const store = useFacturationAppelStore.getState();
      const result = await store.verifierCloture();

      expect(result).toBe(true);
      const state = useFacturationAppelStore.getState();
      expect(state.cloture?.cloture_enCours).toBe(true);
      expect(apiClient.get).toHaveBeenCalledWith('/api/facturation-appel/cloture-status');
    });

    it('should handle API error and return false', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      vi.mocked(apiClient.get).mockRejectedValueOnce(new Error('Network error'));

      const store = useFacturationAppelStore.getState();
      const result = await store.verifierCloture();

      expect(result).toBe(false);
      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe('Network error');
      expect(state.cloture).toBe(null);
    });
  });

  describe('debloquerCloture', () => {
    it('should debloquer cloture in mock mode', async () => {
      useFacturationAppelStore.setState({
        cloture: { cloture_enCours: true, testReseau: 'ERREUR' },
      });

      const store = useFacturationAppelStore.getState();
      await store.debloquerCloture();

      const state = useFacturationAppelStore.getState();
      expect(state.cloture?.cloture_enCours).toBe(false);
      expect(state.isLoading).toBe(false);
      expect(state.error).toBe(null);
    });

    it('should debloquer cloture via API', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      useFacturationAppelStore.setState({
        cloture: { cloture_enCours: true, testReseau: 'ERREUR' },
      });

      const mockResponse: DebloquerClotureResponse = {
        success: true,
        data: { success: true },
      };
      vi.mocked(apiClient.post).mockResolvedValueOnce({ data: mockResponse });

      const store = useFacturationAppelStore.getState();
      await store.debloquerCloture();

      const state = useFacturationAppelStore.getState();
      expect(state.cloture?.cloture_enCours).toBe(false);
      expect(apiClient.post).toHaveBeenCalledWith('/api/facturation-appel/debloquer-cloture');
    });

    it('should handle API error', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      useFacturationAppelStore.setState({
        cloture: { cloture_enCours: true, testReseau: 'ERREUR' },
      });
      vi.mocked(apiClient.post).mockRejectedValueOnce(new Error('Déblocage impossible'));

      const store = useFacturationAppelStore.getState();
      await store.debloquerCloture();

      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe('Déblocage impossible');
      expect(state.isLoading).toBe(false);
    });
  });

  describe('marquerGratuit', () => {
    it('should marquer appel gratuit in mock mode', async () => {
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_1],
      });

      const store = useFacturationAppelStore.getState();
      await store.marquerGratuit(1, 'Problème ligne technique');

      const state = useFacturationAppelStore.getState();
      const updatedAppel = state.historiqueAppels.find((a) => a.id === 1);
      expect(updatedAppel?.gratuite).toBe(true);
      expect(updatedAppel?.raisonGratuite).toBe('Problème ligne technique');
      expect(state.error).toBe(null);
    });

    it('should marquer appel gratuit via API', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_1],
      });

      const mockResponse: MarquerGratuitResponse = {
        success: true,
        data: { success: true },
      };
      vi.mocked(apiClient.put).mockResolvedValueOnce({ data: mockResponse });

      const store = useFacturationAppelStore.getState();
      await store.marquerGratuit(1, 'Urgence médicale');

      const state = useFacturationAppelStore.getState();
      const updatedAppel = state.historiqueAppels.find((a) => a.id === 1);
      expect(updatedAppel?.gratuite).toBe(true);
      expect(updatedAppel?.raisonGratuite).toBe('Urgence médicale');
      expect(apiClient.put).toHaveBeenCalledWith(
        '/api/facturation-appel/marquer-gratuit/1',
        { raison: 'Urgence médicale' }
      );
    });

    it('should handle API error', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_1],
      });
      vi.mocked(apiClient.put).mockRejectedValueOnce(new Error('Modification impossible'));

      const store = useFacturationAppelStore.getState();
      await store.marquerGratuit(1, 'Raison');

      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe('Modification impossible');
    });
  });

  describe('annulerFacturation', () => {
    it('should annuler facturation in mock mode', async () => {
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_2],
      });

      const store = useFacturationAppelStore.getState();
      await store.annulerFacturation(2);

      const state = useFacturationAppelStore.getState();
      const updatedAppel = state.historiqueAppels.find((a) => a.id === 2);
      expect(updatedAppel?.facture).toBe(false);
      expect(updatedAppel?.operationId).toBeUndefined();
      expect(state.error).toBe(null);
    });

    it('should annuler facturation via API', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_2],
      });

      const mockResponse: AnnulerFacturationResponse = {
        success: true,
        data: { success: true },
      };
      vi.mocked(apiClient.delete).mockResolvedValueOnce({ data: mockResponse });

      const store = useFacturationAppelStore.getState();
      await store.annulerFacturation(2);

      const state = useFacturationAppelStore.getState();
      const updatedAppel = state.historiqueAppels.find((a) => a.id === 2);
      expect(updatedAppel?.facture).toBe(false);
      expect(updatedAppel?.operationId).toBeUndefined();
      expect(apiClient.delete).toHaveBeenCalledWith('/api/facturation-appel/annuler/2');
    });

    it('should handle API error', async () => {
      useDataSourceStore.setState({ isRealApi: true });
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_2],
      });
      vi.mocked(apiClient.delete).mockRejectedValueOnce(new Error('Annulation refusée'));

      const store = useFacturationAppelStore.getState();
      await store.annulerFacturation(2);

      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe('Annulation refusée');
    });
  });

  describe('filters', () => {
    it('should set filter societe', () => {
      const store = useFacturationAppelStore.getState();

      store.setFilterSociete('SOC2');

      const state = useFacturationAppelStore.getState();
      expect(state.filterSociete).toBe('SOC2');
    });

    it('should set filter date debut', () => {
      const store = useFacturationAppelStore.getState();
      const date = new Date('2026-02-15');

      store.setFilterDateDebut(date);

      const state = useFacturationAppelStore.getState();
      expect(state.filterDateDebut).toEqual(date);
    });

    it('should set filter date fin', () => {
      const store = useFacturationAppelStore.getState();
      const date = new Date('2026-02-20');

      store.setFilterDateFin(date);

      const state = useFacturationAppelStore.getState();
      expect(state.filterDateFin).toEqual(date);
    });

    it('should reset all filters', () => {
      const store = useFacturationAppelStore.getState();
      store.setFilterSociete('SOC1');
      store.setFilterDateDebut(new Date('2026-02-15'));
      store.setFilterDateFin(new Date('2026-02-20'));

      store.resetFilters();

      const state = useFacturationAppelStore.getState();
      expect(state.filterSociete).toBe('');
      expect(state.filterDateDebut).toBe(null);
      expect(state.filterDateFin).toBe(null);
    });
  });

  describe('error management', () => {
    it('should set error', () => {
      const store = useFacturationAppelStore.getState();

      store.setError('Custom error message');

      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe('Custom error message');
    });

    it('should clear error', () => {
      const store = useFacturationAppelStore.getState();
      store.setError('Error');

      store.setError(null);

      const state = useFacturationAppelStore.getState();
      expect(state.error).toBe(null);
    });
  });

  describe('reset', () => {
    it('should reset store to initial state', () => {
      const store = useFacturationAppelStore.getState();
      store.setFilterSociete('SOC1');
      store.setError('Error');
      useFacturationAppelStore.setState({
        historiqueAppels: [MOCK_APPEL_1],
        coefficientTelephone: 0.15,
        cloture: { cloture_enCours: true, testReseau: 'OK' },
      });

      store.reset();

      const state = useFacturationAppelStore.getState();
      expect(state.historiqueAppels).toEqual([]);
      expect(state.coefficientTelephone).toBe(null);
      expect(state.cloture).toBe(null);
      expect(state.isLoading).toBe(false);
      expect(state.error).toBe(null);
      expect(state.filterSociete).toBe('');
      expect(state.filterDateDebut).toBe(null);
      expect(state.filterDateFin).toBe(null);
    });
  });
});
```
