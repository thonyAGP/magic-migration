const detectAnomalies = (details: DeviseConversion[]): Alert[] => {
  const alerts: Alert[] = [];
  
  details.forEach((d) => {
    if (d.tauxChange <= 0) {
      alerts.push({
        type: 'INVALID_RATE',
        devise: d.devise,
        message: `Taux de change invalide pour ${d.devise}`,
      });
    }
    if (d.montant < 0) {
      alerts.push({
        type: 'NEGATIVE_AMOUNT',
        devise: d.devise,
        message: `Montant négatif pour ${d.devise}`,
      });
    }
    if (d.montant === 0) {
      alerts.push({
        type: 'ZERO_AMOUNT',
        devise: d.devise,
        message: `Montant nul pour ${d.devise}`,
      });
    }
  });
  
  return alerts;
};
