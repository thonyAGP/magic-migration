```typescript
import type { DeversementResult } from "@/types/deversementTransaction";
import { cn } from "@/lib/utils";

interface ResultatDeversementPanelProps {
  result: DeversementResult | null;
  numeroTicket?: number | null;
  error?: string | null;
  className?: string;
}

export const ResultatDeversementPanel = ({
  result,
  numeroTicket,
  error,
  className,
}: ResultatDeversementPanelProps) => {
  if (error) {
    return (
      <div className={cn("bg-red-50 border border-red-200 rounded-lg p-6", className)}>
        <div className="flex items-start gap-3">
          <div className="flex-shrink-0 w-8 h-8 bg-red-100 rounded-full flex items-center justify-center">
            <svg className="w-5 h-5 text-red-600" fill="none" viewBox="0 0 24 24" stroke="currentColor">
              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" />
            </svg>
          </div>
          <div className="flex-1">
            <h3 className="text-sm font-semibold text-red-800 mb-1">Erreur de déversement</h3>
            <p className="text-sm text-red-700">{error}</p>
          </div>
        </div>
      </div>
    );
  }

  if (!result) {
    return null;
  }

  return (
    <div className={cn("bg-green-50 border border-green-200 rounded-lg p-6", className)}>
      <div className="flex items-start gap-3 mb-4">
        <div className="flex-shrink-0 w-8 h-8 bg-green-100 rounded-full flex items-center justify-center">
          <svg className="w-5 h-5 text-green-600" fill="none" viewBox="0 0 24 24" stroke="currentColor">
            <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M5 13l4 4L19 7" />
          </svg>
        </div>
        <div className="flex-1">
          <h3 className="text-sm font-semibold text-green-800 mb-1">Déversement effectué avec succès</h3>
          <p className="text-sm text-green-700">La transaction a été traitée correctement</p>
        </div>
      </div>

      <div className="grid grid-cols-1 md:grid-cols-2 gap-4 mb-4">
        <div className="bg-white rounded border border-green-100 p-4">
          <div className="text-xs font-medium text-gray-500 mb-1">Vente ID</div>
          <div className="text-lg font-semibold text-gray-900">{result.venteId}</div>
        </div>

        {numeroTicket !== null && numeroTicket !== undefined && (
          <div className="bg-white rounded border border-green-100 p-4">
            <div className="text-xs font-medium text-gray-500 mb-1">Numéro de ticket</div>
            <div className="text-lg font-semibold text-gray-900">{numeroTicket}</div>
          </div>
        )}

        {result.numeroTicket !== undefined && (
          <div className="bg-white rounded border border-green-100 p-4">
            <div className="text-xs font-medium text-gray-500 mb-1">Ticket généré</div>
            <div className="text-lg font-semibold text-gray-900">{result.numeroTicket}</div>
          </div>
        )}

        {result.transfertAffectationId !== undefined && (
          <div className="bg-white rounded border border-green-100 p-4">
            <div className="text-xs font-medium text-gray-500 mb-1">ID transfert affectation</div>
            <div className="text-lg font-semibold text-gray-900">{result.transfertAffectationId}</div>
          </div>
        )}
      </div>

      <div className="bg-white rounded border border-green-100 p-4 mb-4">
        <h4 className="text-xs font-medium text-gray-500 mb-3">Compte GM</h4>
        <div className="grid grid-cols-2 gap-4">
          <div>
            <div className="text-xs text-gray-500 mb-1">Ancien solde</div>
            <div className="text-base font-semibold text-gray-900">
              {result.compteGMAncienSolde.toFixed(2)} €
            </div>
          </div>
          <div>
            <div className="text-xs text-gray-500 mb-1">Nouveau solde</div>
            <div className="text-base font-semibold text-green-700">
              {result.compteGMNouveauSolde.toFixed(2)} €
            </div>
          </div>
        </div>
      </div>

      {result.operationsDiverses.length > 0 && (
        <div className="bg-white rounded border border-green-100 p-4">
          <h4 className="text-xs font-medium text-gray-500 mb-3">
            Opérations diverses créées ({result.operationsDiverses.length})
          </h4>
          <div className="space-y-2 max-h-48 overflow-y-auto">
            {result.operationsDiverses.map((od) => (
              <div key={od.id} className="flex items-center justify-between py-2 border-b border-gray-100 last:border-0">
                <div className="flex-1">
                  <div className="text-sm font-medium text-gray-900">{od.typeOD.toUpperCase()}</div>
                  {od.description && (
                    <div className="text-xs text-gray-500">{od.description}</div>
                  )}
                </div>
                <div className="text-sm font-semibold text-gray-900">
                  {od.montant.toFixed(2)} €
                </div>
              </div>
            ))}
          </div>
        </div>
      )}
    </div>
  );
};
```
