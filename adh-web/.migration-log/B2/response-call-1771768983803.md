```tsx
import { memo } from 'react';
import { Button, Dialog } from '@/components/ui';
import type { SessionConcurrency } from '@/types/sessionConcurrency';

interface ConflictWarningPanelProps {
  open: boolean;
  conflictingSession: SessionConcurrency | null;
  onCancel: () => void;
  onForceOpen: () => void;
  isLoading?: boolean;
}

export const ConflictWarningPanel = memo<ConflictWarningPanelProps>(({
  open,
  conflictingSession,
  onCancel,
  onForceOpen,
  isLoading = false,
}) => {
  if (!conflictingSession) return null;

  const formattedTimestamp = new Date(conflictingSession.timestamp).toLocaleString('fr-FR', {
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
    hour: '2-digit',
    minute: '2-digit',
    second: '2-digit',
  });

  return (
    <Dialog open={open} onClose={onCancel}>
      <div className="bg-white rounded-lg shadow-xl p-6 max-w-lg w-full">
        <div className="flex items-start gap-3 mb-4">
          <div className="flex-shrink-0 w-10 h-10 rounded-full bg-yellow-100 flex items-center justify-center">
            <svg className="w-6 h-6 text-yellow-600" fill="none" viewBox="0 0 24 24" stroke="currentColor">
              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 9v2m0 4h.01m-6.938 4h13.856c1.54 0 2.502-1.667 1.732-3L13.732 4c-.77-1.333-2.694-1.333-3.464 0L3.34 16c-.77 1.333.192 3 1.732 3z" />
            </svg>
          </div>
          <div className="flex-1">
            <h3 className="text-lg font-semibold text-gray-900 mb-1">
              Session déjà ouverte
            </h3>
            <p className="text-sm text-gray-600">
              Une session est déjà active pour ce compte. Forcer l'ouverture fermera automatiquement la session existante.
            </p>
          </div>
        </div>

        <div className="bg-gray-50 rounded-lg p-4 mb-6 space-y-2">
          <div className="flex justify-between items-center">
            <span className="text-sm font-medium text-gray-700">Terminal:</span>
            <span className="text-sm text-gray-900 font-mono">{conflictingSession.terminalId}</span>
          </div>
          <div className="flex justify-between items-center">
            <span className="text-sm font-medium text-gray-700">Société:</span>
            <span className="text-sm text-gray-900">{conflictingSession.societe}</span>
          </div>
          <div className="flex justify-between items-center">
            <span className="text-sm font-medium text-gray-700">Compte:</span>
            <span className="text-sm text-gray-900">{conflictingSession.compte}</span>
          </div>
          <div className="flex justify-between items-center">
            <span className="text-sm font-medium text-gray-700">Filiation:</span>
            <span className="text-sm text-gray-900">{conflictingSession.filiation}</span>
          </div>
          <div className="flex justify-between items-center">
            <span className="text-sm font-medium text-gray-700">Ouvert le:</span>
            <span className="text-sm text-gray-900">{formattedTimestamp}</span>
          </div>
          {conflictingSession.codeCalcul && (
            <div className="flex justify-between items-center">
              <span className="text-sm font-medium text-gray-700">Code calcul:</span>
              <span className="text-sm text-gray-900">{conflictingSession.codeCalcul}</span>
            </div>
          )}
          {conflictingSession.coffreEnCoursComptage && (
            <div className="flex items-center gap-2 pt-2 border-t border-gray-200">
              <svg className="w-4 h-4 text-orange-600" fill="currentColor" viewBox="0 0 20 20">
                <path fillRule="evenodd" d="M8.257 3.099c.765-1.36 2.722-1.36 3.486 0l5.58 9.92c.75 1.334-.213 2.98-1.742 2.98H4.42c-1.53 0-2.493-1.646-1.743-2.98l5.58-9.92zM11 13a1 1 0 11-2 0 1 1 0 012 0zm-1-8a1 1 0 00-1 1v3a1 1 0 002 0V6a1 1 0 00-1-1z" clipRule="evenodd" />
              </svg>
              <span className="text-sm font-medium text-orange-700">Coffre en cours de comptage</span>
            </div>
          )}
        </div>

        <div className="flex gap-3 justify-end">
          <Button
            variant="secondary"
            onClick={onCancel}
            disabled={isLoading}
          >
            Annuler
          </Button>
          <Button
            variant="primary"
            onClick={onForceOpen}
            disabled={isLoading}
            className="bg-yellow-600 hover:bg-yellow-700 focus:ring-yellow-500"
          >
            {isLoading ? 'Traitement...' : 'Forcer l\'ouverture'}
          </Button>
        </div>
      </div>
    </Dialog>
  );
});

ConflictWarningPanel.displayName = 'ConflictWarningPanel';
```
